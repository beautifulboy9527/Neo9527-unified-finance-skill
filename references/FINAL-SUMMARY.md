## 💡 使用示例

### 统一入口 CLI

```bash
# 行情查询
python finance.py quote AAPL
python finance.py quote 600519
python finance.py quote 00700.HK

# 图表生成
python finance.py chart AAPL 3mo --rsi --macd --bb
python finance.py chart 600519 6mo --rsi --macd

# 行业热力图
python finance.py heatmap ab

# 资金流向
python finance.py fundflow 600519

# 警报管理
python finance.py alert add AAPL --target 260 --stop 240
python finance.py alert list
python finance.py alert check

# 完整报告
python complete_report.py AAPL 6mo

# 链式流程
python pipeline.py AAPL deep
python pipeline.py 600519 daily
```

### Python API

```python
# 行情查询
from data_fetcher import get_quote
data = get_quote('AAPL')

# 图表生成
from chart_generator import generate_chart
path = generate_chart('600519', '6mo', {'rsi': True, 'macd': True})

# 多数据源
from multi_source_data import MultiSourceData
msd = MultiSourceData('AAPL')
data = msd.get_quote()
validation = msd.validate_data(data)

# 链式流程
from pipeline import create_deep_analysis_pipeline
pipeline = create_deep_analysis_pipeline()
result = pipeline.run({'symbol': 'AAPL'})
```
